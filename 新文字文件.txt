package course07;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class ch7_4StringLengthCountSortDesc{

	public static void main(String[] args) {
		//先取檔案的路徑
		Path fielpath=Paths.get(
				System.getProperty("user.dir")+"\\src\\resources\\dict\\web2");
		
		System.out.println(fielpath);
		//H:\小杰\其他知識\JAVA\現代JAVA\JayPracticeJava8\JaysJava8Practice\src\main\resources\dict\web2
		//H:\小杰\其他知識\JAVA\現代JAVA\JayPracticeJava8\JaysJava8Practice\src\resources\dict
	
		//從字典中找出單字長度大於20的，並依字串長度分群統計各有多個單字
	    try (Stream<String> words = Files.lines(fielpath)) {
            words.filter(s -> s.length() > 20)
                    .collect(Collectors.groupingBy(String::length,Collectors.counting()))
                    .forEach((len,num)->System.out.println(len+": "+num));
        } catch (IOException e) {
            e.printStackTrace();
        }
	}

}
